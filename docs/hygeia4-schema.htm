Table admins
aid INT NOT NULL AUTOINCREMENT
email TEXT NOT NULL
hpwd TEXT NOT NULL
Indexes
pk_admins primary key ( aid)


Table components
cid INT NOT NULL AUTOINCREMENT
mid INT NOT NULL
fid INT NOT NULL
count DOUBLE
Indexes
pk_components primary key ( cid)
fid ( fid)
mid ( mid)
Foreign Keys
components_ibfk_2 ( mid ) ref meals ( mid )
components_ibfk_1 ( fid ) ref foods ( fid )


Table favorites
fid INT NOT NULL AUTOINCREMENT
mid INT NOT NULL
uid INT NOT NULL
Indexes
pk_favorites primary key ( fid)
uid ( uid)
mid ( mid)
Foreign Keys
favorites_ibfk_2 ( mid ) ref meals ( mid )
favorites_ibfk_1 ( uid ) ref users ( uid )
